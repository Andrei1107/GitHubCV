void bisectie(double x1, double x2, double eps, int n, double (*f)(double), double *sol, int *er)
{
	*er=0;
	while((*er==0)&&(n>0))
	{
		*sol=(x1+x2)/2;
		if((*f)(*sol)==0)
			*er=1;
		else
		{
			if(x2-x1<eps)
				*er=2;
			else
				if((*f)(*sol)*(*f)(x1)<0)
					x2=*sol;
				else
					x1=*sol;
		}
		n--;
	}
}
